layerDepth = depthsp[1];
}else{
layerDepth = depthsp[i] - depthsp[i-1];
}
CumLayerDepth += layerDepth;
if(rootDepth > CumLayerDepth){
CumRootDist++;
}
}
for(j=0;j<layer;j++){
if(j < CumRootDist){
a = dpois(j+1,CumRootDist*rfl,0);
rootDist[j] = a;
ca += a;
}else{
rootDist[j] = 0;
}
}
for(k=0;k<layer;k++){
tmp.rootDist[k] = rootDist[k] / ca;
}
return(tmp);
}
